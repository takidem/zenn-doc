# Vim関連更新

## vim関連を更新したので、何となく投稿

### .vimrc

```bash
syntax on
set completeopt=menuone,noinsert
set nocompatible
set whichwrap=b,s,h,l,<,>,[,]
set backspace=indent,eol,start
set number
set relativenumber
set tabstop=2
set shiftwidth=2
set expandtab
set autoindent
set smartindent
set hlsearch
set incsearch
set ignorecase
set smartcase
set cursorline
set cursorcolumn
set showmatch
set wildmenu
set wildmode=list:longest
set background=dark
set termguicolors
set cmdheight=2
set laststatus=2
set ruler
set splitbelow
set splitright
set wrap
set linebreak
set swapfile
set foldmethod=syntax
set belloff=all
set foldlevelstart=99
set clipboard+=unnamed,unnamedplus
set updatetime=300
set encoding=utf-8
set conceallevel=0
filetype plugin indent on
 
" NERDTreeToggle用キーマップ
inoremap <silent><expr> <c-space> coc#refresh()
nnoremap <C-t> :NERDTreeToggle<CR>
nnoremap <C-x> :CreateCompletion<CR>
inoremap <C-x> <Esc>li<C-g>u<Esc>l:CreateCompletion<CR>
let NERDTreeShowHidden=1
let mapleader = " "
let s:MINIMUM_COMPLETE_LENGTH = 3

" キーマップ
inoremap <silent> jj <ESC>
noremap <silent> <ESC><ESC> :<C-u>nohlsearch<CR>
nnoremap <C-f> :FloatermToggle<CR>
tnoremap <Esc> <C-\><C-n>:FloatermToggle<CR>

" ウィンドウ変更時キーマップ
nnoremap <leader>l $
nnoremap <leader>h 0
nnoremap wj <C-w>j
nnoremap wk <C-w>k
nnoremap wl <C-w>l
nnoremap wh <C-w>h
let g:ycm_language_server = []
let g:indentLine_color_term = 239
let g:indentLine_conceallevel = 0
let g:previm_wsl_mode = 1
let g:lsp_diagnostics_echo_cursor = 1
let g:sclow_block_buftypes = ['terminal', 'prompt']
let g:sclow_hide_full_length = 1
let g:sclow_sbar_text = '┃'
let g:vim_json_syntax_conceal = 0
inoremap <expr> <Tab>   pumvisible() ? "\<C-n>" : "\<Tab>"
inoremap <expr> <S-Tab> pumvisible() ? "\<C-p>" : "\<S-Tab>"
inoremap <expr> <cr>    pumvisible() ? asyncomplete#close_popup() : "\<cr>"
 
" COC用キーマップ
nmap <silent> gd <Plug>(coc-definition)
inoremap <silent><expr> <c-space> coc#refresh()
inoremap <silent><expr> <tab> pumvisible() ? coc#_select_confirm()
      \: "\<C-g>u\<CR>\<c-r>=coc#on_enter()\<CR>"
nmap <silent> <C-w>gd <cmd>vs<cr><Plug>(coc-definition)
nmap <silent><nowait> gd <Plug>(coc-definition)
nmap <silent><nowait> gy <Plug>(coc-type-definition)
nmap <silent><nowait> gi <Plug>(coc-implementation)
nmap <silent><nowait> gr <Plug>(coc-references)
xmap <leader>f  <Plug>(coc-format-selected)
nmap <leader>f  <Plug>(coc-format-selected)
nnoremap <silent> K :call ShowDocumentation()<CR>
let g:ale_linters = {
    \ 'sh': ['language_server'],
    \ }
function! CheckBackspace() abort
  let col = col('.') - 1
  return !col || getline('.')[col - 1]  =~# '\s'
endfunction
function! ShowDocumentation()
  if CocAction('hasProvider', 'hover')
    call CocActionAsync('doHover')
  else
    call feedkeys('K', 'in')
  endif
endfunction
nmap <f2> <Plug>(coc-rename)
nmap <silent> [g <Plug>(coc-diagnostic-prev)
nmap <silent> ]g <Plug>(coc-diagnostic-next)
nmap <esc><esc> <cmd>call coc#float#close_all()<cr>
nnoremap <silent><nowait> <leader>o  :<C-u>CocList outline<cr>
nnoremap <silent><nowait> <leader>s  :<C-u>CocList -I symbols<cr>
nnoremap <silent> gh :call <SID>show_documentation()<CR>
nnoremap <silent> <leader>a <cmd>CocAction<cr>
function! s:show_documentation()
    if (index(['vim','help'], &filetype) >= 0)
      execute 'h '.expand('<cword>')
    elseif (coc#rpc#ready())
      call CocActionAsync('doHover')
    else
      execute '!' . &keywordprg . " " . expand('<cword>')
    endif
  endfunction
  " higlight when hover cursor
  autocmd CursorHold * silent call CocActionAsync('highlight')
 
" プラグイン管理
call plug#begin('~/.vim/plugged')
  Plug 'preservim/nerdtree'
  Plug 'prabirshrestha/async.vim'
  Plug 'carlsmedstad/vim-bicep'
  Plug 'leafgarland/typescript-vim'
  Plug 'prabirshrestha/vim-lsp'
  Plug 'prabirshrestha/asyncomplete.vim'
  Plug 'prabirshrestha/asyncomplete-lsp.vim'
  Plug 'mattn/vim-lsp-settings'
  Plug 'preservim/vim-markdown'
  Plug 'pangloss/vim-javascript'
  Plug 'hashivim/vim-terraform' , { 'for': 'terraform'}
  Plug 'vim-syntastic/syntastic'
  Plug 'juliosueiras/vim-terraform-completion'
  Plug 'neoclide/coc.nvim', {'branch': 'release'}
  Plug 'vim-airline/vim-airline'
  Plug 'jistr/vim-nerdtree-tabs'
  Plug 'Yggdroot/indentLine'
  Plug 'elzr/vim-json'
  Plug 'tpope/vim-surround'
  Plug 'gosukiwi/vim-smartpairs'
  Plug 'tpope/vim-fugitive'
  Plug 'tpope/vim-repeat'
  Plug 'tpope/vim-commentary'
  Plug 'ryanoasis/vim-devicons'
  Plug 'ConradIrwin/vim-bracketed-paste'
  Plug 'previm/previm'
  Plug 'obcat/vim-hitspop'
  Plug 'obcat/vim-sclow'
  Plug 'psliwka/vim-smoothie'
  Plug 'simeji/winresizer'
  Plug 'mattn/vim-maketable'
  Plug 'junegunn/fzf', { 'do': { -> fzf#install() } }
  Plug 'junegunn/fzf.vim'
  Plug 'voldikss/vim-floaterm'
  Plug 'sainnhe/everforest'
call plug#end()
 
" colorscheme
if has('termguicolors')
  set termguicolors
endif
 
set background=dark
let g:everforest_background = 'hard'
 
colorscheme everforest
let g:lightline = {'colorscheme' : 'everforest'}

" terraformLSP設定
if executable('terraform-ls')
  au User lsp_setup call lsp#register_server({
    \ 'name': 'terraform-ls',
    \ 'cmd': {server_info->['terraform-ls']},
    \ 'whitelist': ['terraform','tf'],
    \ })
endif
```
